### 1.数组去重

```javascript
//es5数组去重
function unique(arr){
    var res =[];
    for(var i=0,len=arr.length;i<len;i++){
        var obj = arr[i];
        if(res.indexOf(obj)===-1) res.push(obj);
    }
    return res;
}

//es6数组去重
var array = [1, 2, 1, 1, '1'];
function unique(array) {
   return Array.from(new Set(array));
}
console.log(unique(array)); // [1, 2, "1"]

//再简化下：
function unique(array) {
    return [...new Set(array)];
}
//再简化下：
var unique = (a) => [...new Set(a)]

```
